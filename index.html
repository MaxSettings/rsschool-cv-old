      function decoder(sequence) {
        let res = [];
        let zeroesCount = 0;

        for (let i = 0; i < sequence.length; i++) {
          if (sequence[i] === "0") {
            zeroesCount++;
          } else {
            res.push(parseInt(sequence.slice(i, i + zeroesCount + 1), 2) - 1);
            i += zeroesCount;
            zeroesCount = 0;
          }
        }

      }
